using System;

class Phone
{
    private string brand;
    private int batteryLevel;

    public string Brand { get { return brand; } set { brand = value; } }

    public int BatteryLevel
    {
        get { return batteryLevel; }
        set
        {
            if (value >= 0 && value <= 100)
                batteryLevel = value;
            else
                Console.WriteLine("Заряд должен быть от 0 до 100!");
        }
    }

    public Phone(string brand, int batteryLevel)
    {
      
        this.brand = brand;
        batteryLevel = batteryLevel >= 0 && batteryLevel <= 100 ? batteryLevel : 0;
    }

    public void UsePhone()
    {
        if (batteryLevel >= 10)
        {
            batteryLevel = 10;
        }
        else
        {
            batteryLevel = 0; 
        }

        
        Console.WriteLine($"Телефон {brand}, заряд: {batteryLevel}%.");
    }
}

class Program
{
    static void Main()
    {
        Phone myPhone = new Phone("Samsung", 0);

       
        myPhone.UsePhone(); 
        myPhone.UsePhone(); 

        myPhone.BatteryLevel = -10; 

        
        

        Console.ReadLine();
    }
}
